age int:=&age;
begin
if age>=18 then
dbms_output.put_line('You are an adult');
else

end if;
end;

declare
pr float:=&principal_amount;
tm float:=&time_in_years;
rt float:=&rate_of_interest;
si float;
begin
si:=(pr*rt*tm)/100;
dbms_output.put_line('Simple interest is : '||si);
end;
